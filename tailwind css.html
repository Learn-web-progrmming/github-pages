مقدمة في Tailwind CSS

تعرف على أساسيات Tailwind CSS وكيفية إعداده في مشروعك. 

<!-- إضافة Tailwind CSS عبر CDN -->
<link href="https://cdn.jsdelivr.net/npm/tailwindcss@2.2.19/dist/tailwind.min.css" rel="stylesheet">

<!-- أو التثبيت عبر npm -->
npm install -D tailwindcss
npx tailwindcss init

<!-- تهيئة ملف tailwind.config.js -->
module.exports = {
  content: ["./src/**/*.{html,js}"],
  theme: {
    extend: {},
  },
  plugins: [],
}

<!-- إنشاء ملف CSS رئيسي -->
@tailwind base;
@tailwind components;
@tailwind utilities;

هذه مقدمة نظرية حول إعداد Tailwind CSS، لا يوجد ناتج مرئي لهذا الدرس.

Flex و Grid

استخدام أنظمة Flexbox و Grid لإنشاء تخطيطات متقدمة. 

<!-- Flexbox مثال -->
<div class="flex flex-wrap justify-between">
    <div class="w-1/4 p-4 bg-purple-500 m-2">1</div>
    <div class="w-1/4 p-4 bg-purple-500 m-2">2</div>
    <div class="w-1/4 p-4 bg-purple-500 m-2">3</div>
    <div class="w-1/4 p-4 bg-purple-500 m-2">4</div>
</div>

<!-- Grid مثال -->
<div class="grid grid-cols-3 gap-4 mt-6">
    <div class="bg-pink-500 p-4">A</div>

التخصيص والإعداد

كيفية تخصيص Tailwind ليتناسب مع احتياجات مشروعك. 

// tailwind.config.js
module.exports = {
  theme: {
    extend: {
      colors: {
        primary: '#8B5CF6',
        secondary: '#EC4899',
      },
      fontFamily: {
        arabic: ['"Tajawal"', 'sans-serif'],
      },
    },
  },
  variants: {
    extend: {
      backgroundColor: ['active'],
      textColor: ['active'],
    },
  },
  plugins: [
    require('@tailwindcss/forms'),
    require('@tailwindcss/typography'),
  ],
}

هذا مثال لتخصيص ملف tailwind.config.js، لا يوجد ناتج مرئي لهذا الدرس.

أفضل الممارسات

نصائح وأفضل الممارسات لاستخدام Tailwind CSS بكفاءة. 

<!-- 1. استخدم @apply للمكونات المتكررة -->
<style>
    .btn {
        @apply px-4 py-2 rounded font-medium 
               transition-colors duration-200;
    }
    .btn-primary {
        @apply btn bg-purple-600 text-white 
               hover:bg-purple-700;
    }
</style>

<!-- 2. استخدم المكونات البرمجية للمكونات المعقدة -->
// React/Vue/etc component
function Card({ title, content }) {
    return (
        <div className="bg-white rounded-lg shadow p-6">
            <h3 className="text-xl font-bold mb-2">
                {title}
            </h3>
            <p className="text-gray-600">{content}</p>
        </div>
    );
}

<!-- 3. استخدم PurgeCSS لإزالة الفئات غير المستخدمة -->
// tailwind.config.js
module.exports = {
  purge: ['./src/**/*.{html,js,jsx,ts,tsx}'],
  // ...
}

هذه أفضل ممارسات لاستخدام Tailwind، لا يوجد ناتج مرئي لهذا الدرس.
