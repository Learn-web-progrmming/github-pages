مقدمة في React

تعرف على أساسيات React وكيفية إنشاء مكون بسيط باستخدام JSX.


import React from 'react';

function Welcome() {
  return (
    <h1>أهلاً بك في React!</h1>
  );
}

export default Welcome;

المكونات

تعرف على كيفية إنشاء مكونات React القابلة لإعادة الاستخدام.


function Greeting(props) {
  return <h1>مرحباً، {props.name}!</h1>;
}

Props

كيفية تمرير البيانات إلى المكونات باستخدام الـProps.


function Welcome(props) {
  return <h1>مرحباً، {props.name}!</h1>;
}

State

تعرف على مفهوم الحالة (State) وكيفية التحكم فيها.


import { useState } from 'react';

function Counter() {
  const [count, setCount] = useState(0);

  return (
    <div>
      <p>العدد: {count}</p>
      <button onClick={() => setCount(count + 1)}>زيادة</button>
    </div>
  );
}
